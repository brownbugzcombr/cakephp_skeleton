  220. /**
  221.  * Setup the behavior. It stores a reference to the model, merges the default options with the options for each field, and setup the validation rules.
  222.  *

  224.  * @param array $settings Settings (optional)
  225.  * @return void

  228.     function setup(&$model, $settings = array()) {
  229.         $this->__fields[$model->alias] = array();
  230.         foreach ($settings as $field => $options) {
  231.             // Check if they even PASSED IN parameters
  232.             if (!is_array($options)) {
  233.                 // You jerks!
  234.                 $field = $options;
  235.                 $options = array();
  236.             }
  237.  
  238.             // Inherit model's lack of table use if not set in options
  239.             // regardless of whether or not we set the table option
  240.             if (!$model->useTable) {
  241.                 $options['useTable'] = false;
  242.             }
  243.  
  244.             // Merge given options with defaults
  245.             $options = $this->_arrayMerge($this->defaultOptions, $options);
  246.  
  247.             // Check if given field exists
  248.             if ($options['useTable'] && !$model->hasField($field)) {
  249.                 trigger_error(sprintf(__d('meio_upload', 'MeioUploadBehavior Error: The field "%s" doesn\'t exists in the model "%s".', true), $field, $model->alias), E_USER_WARNING);
  250.             }
  251.  
  252.             // Including the default name to the replacements
  253.             if ($options['default']) {
  254.                 if (strpos($options['default'], '.') === false) {
  255.                     trigger_error(__d('meio_upload', 'MeioUploadBehavior Error: The default option must be the filename with extension.', true), E_USER_ERROR);
  256.                 }
  257.                 $this->_includeDefaultReplacement($options['default']);
  258.             }
  259.  
  260.             // Verifies if the thumbsizes names is alphanumeric
  261.             if ($options['thumbnails'] == true) {
  262.                 foreach ($options['thumbsizes'] as $name => $size) {
  263.                     if (empty($name) || !ctype_alnum($name)) {
  264.                         trigger_error(__d('meio_upload', 'MeioUploadBehavior Error: The thumbsizes names must be alphanumeric.', true), E_USER_ERROR);
  265.                     }
  266.                 }
  267.             }
  268.  
  269.             // Process the max_size if it is not numeric
  270.             $options['maxSize'] = $this->_sizeToBytes($options['maxSize']);
  271.  
  272.             // Replace tokens of the dir and field, check it doesn't have a DS on the end
  273.             $tokens = array('{ModelName}', '{fieldName}', '{DS}', '/', '\\');
  274.             $options['dir'] = rtrim($this->_replaceTokens($model, $options['dir'], $field, $tokens), DS);
  275.             $options['uploadName'] = rtrim($this->_replaceTokens($model, $options['uploadName'], $field, $tokens), DS);
  276.  
  277.             // Create the folders for the uploads only if you want
  278.             if($options['createDirectory']) {
  279.                 // Create the folders for the uploads
  280.                 if (!empty($options['thumbsizes'])) {
  281.                     $this->_createFolders($options['dir'], $options['thumbnailDir'], array_keys($options['thumbsizes']));
  282.                 } else {
  283.                     $this->_createFolders($options['dir'], $options['thumbnailDir'], array());
  284.                 }
  285.             }
  286.  
  287.             // Replace tokens in the fields names
  288.             if ($options['useTable']) {
  289.                 foreach ($options['fields'] as $fieldToken => $fieldName) {
  290.                     $options['fields'][$fieldToken] = $this->_replaceTokens($model, $fieldName, $field, $tokens);
  291.                 }
  292.             }
  293.             $this->__fields[$model->alias][$field] = $options;
  294.         }
  295.     }

  370.     function upload(&$model, $data) {
  371.         $result = $this->_uploadFile($model, $data);
  372.         if (is_bool($result)) {
  373.             return $result;
  374.         } elseif (is_array($result)) {
  375.             if ($result['return'] === false) {
  376.                 // Upload failed, lets see why
  377.                 switch($result['reason']) {
  378.                     case 'validation':
  379.                         $model->validationErrors[$result['extra']['field']] = $result['extra']['error'];
  380.                         break;
  381.                 }
  382.                 return false;
  383.             } else {
  384.                 $this->data = $result['data'];
  385.                 return true;
  386.             }
  387.         } else {
  388.             return false;
  389.         }

  432. /**
  433.  * Checks if the folder exists or can be created or writable.

  436.  * @param array $data 
  437.  * @return boolean

  439.  */

  441.         foreach ($data as $fieldName => $field) {
  442.             if (!$model->validate[$fieldName]['Dir']['check']) {
  443.                 continue;
  444.             }
  445.             $options = $this->__fields[$model->alias][$fieldName];
  446.             if (empty($field['remove']) || empty($field['name'])) {
  447.                 // Check if directory exists and create it if required
  448.                 if (!is_dir($options['dir'])) {
  449.                     if ($options['createDirectory']) {
  450.                         $folder = &new Folder();
  451.                         if (!$folder->create($options['dir'])) {

  457.                         return false;
  458.                     }
  459.                 }
  460.  
  461.                 // Check if directory is writable

  463.                     trigger_error(sprintf(__d('meio_upload', 'MeioUploadBehavior Error: The directory %s isn\'t writable.', true), $options['dir']), E_USER_WARNING);

  652. /**
  653.  * Check generic to size of image
  654.  *
  655.  * @param object $model Reference to model
  656.  * @param array $data Reference to data
  657.  * @param string $type Type of check: minHeight, minWidth, maxHeight, maxWidth
  658.  * @return boolean 
  659.  * @access protected
  660.  */
  661.     function _uploadCheckSize(&$model, &$data, $type) {
  662.         foreach ($data as $fieldName => $field) {
  663.             if (!$model->validate[$fieldName][ucfirst($type)]['check'] || empty($field['tmp_name'])) {
  664.                 continue;
  665.             }
  666.             $options = $this->__fields[$model->alias][$fieldName];
  667.             list($imgWidth, $imgHeight) = getimagesize($field['tmp_name']);
  668.             $imgType = 'img' . substr($type, 3);
  669.             if (substr($type, 0, 3) === 'min') {
  670.                 if (!empty($field['name']) && $options['length'][$type] > 0 && $$imgType < $options['length'][$type]) {
  671.                     return false;
  672.                 }
  673.             } else {
  674.                 if (!empty($field['name']) && $options['length'][$type] > 0 && $$imgType > $options['length'][$type]) {
  675.                     return false;
  676.                 }
  677.             }
  678.         }
  679.         return true;
  680.     }

  682. /**
  683.  * Uploads the files
  684.  *
  685.  * @param object $model Reference to model
  686.  * @param array $data Optional Containing data to be saved
  687.  * @return mixed 
  688.  * @access protected
  689.  */
  690.     function _uploadFile(&$model, $data = null) {
  691.         if (!isset($data) || !is_array($data)) {
  692.             $data =& $model->data;
  693.         }
  694.         foreach ($this->__fields[$model->alias] as $fieldName => $options) {
  695.             // Take care of removal flagged field
  696.             // However, this seems to be kind of code duplicating, see line ~711
  697.             if (!empty($data[$model->alias][$fieldName]['remove'])) {
  698.                 $this->_markForDeletion($model, $fieldName, $data, $options['default'], $options['thumbnailDir']);
  699.                 $data = $this->_nullifyDataFields($model, $fieldName, $data, $options);
  700.                 $result = array('return' => true, 'data' => $data);
  701.                 continue;
  702.             }
  703.             // If no file was selected we do not need to proceed
  704.             if (empty($data[$model->alias][$fieldName]['name'])) {
  705.                 unset($data[$model->alias][$fieldName]);
  706.                 $result = array('return' => true, 'data' => $data);
  707.                 continue;
  708.             }
  709.             $pos = strrpos($data[$model->alias][$fieldName]['type'], '/');
  710.             $sub = substr($data[$model->alias][$fieldName]['type'], $pos+1);
  711.             list(,$ext) = $this->_splitFilenameAndExt($data[$model->alias][$fieldName]['name']);
  712.  
  713.             // Put in a subfolder if the user wishes it
  714.             if (isset($options['folderAsField']) && !empty($options['folderAsField']) && is_string($options['folderAsField'])) {
  715.                 $options['dir'] = $options['dir'] . DS . $data[$model->alias][$options['folderAsField']];
  716.                 $this->__fields[$model->alias][$fieldName]['dir'] = $options['dir'];
  717.             }
  718.  
  719.             // Check whether or not the behavior is in useTable mode
  720.             if ($options['useTable'] == false) {
  721.                 $this->_includeDefaultReplacement($options['default']);
  722.                 $this->_fixName($model, $fieldName, false);
  723.                 $saveAs = $options['dir'] . DS . $data[$model->alias][$options['uploadName']] . '.' . $sub;
  724.  
  725.                 // Attempt to move uploaded file
  726.                 $copyResults = $this->_copyFileFromTemp($data[$model->alias][$fieldName]['tmp_name'], $saveAs);
  727.                 if ($copyResults !== true) {
  728.                     $result = array('return' => false, 'reason' => 'validation', 'extra' => array('field' => $fieldName, 'error' => $copyResults));
  729.                     continue;
  730.                 }
  731.  
  732.                 // If the file is an image, try to make the thumbnails
  733.                 if ((count($options['thumbsizes']) > 0) && count($options['allowedExt']) > 0 && in_array($data[$model->alias][$fieldName]['type'], $this->_imageTypes)) {
  734.                     $this->_createThumbnails($model, $data, $fieldName, $saveAs, $ext, $options);
  735.                 }
  736.  
  737.                 $data = $this->_unsetDataFields($model->alias, $fieldName, $model->data, $options);
  738.                 $result = array('return' => true, 'data' => $data);
  739.                 continue;
  740.             } else {
  741.                 // if the file is marked to be deleted, use the default or set the field to null
  742.                 if (!empty($data[$model->alias][$fieldName]['remove'])) {
  743.                     if ($options['default']) {
  744.                         $data[$model->alias][$fieldName] = $options['default'];
  745.                     } else {
  746.                         $data[$model->alias][$fieldName] = null;
  747.                     }
  748.                     //if the record is already saved in the database, set the existing file to be removed after the save is sucessfull
  749.                     if (!empty($data[$model->alias][$model->primaryKey])) {
  750.                         $this->_setFileToRemove($model, $fieldName, $options['thumbnailDir']);
  751.                     }
  752.                 }
  753.  
  754.                 // If no file has been upload, then unset the field to avoid overwriting existant file
  755.                 if (!isset($data[$model->alias][$fieldName]) || !is_array($data[$model->alias][$fieldName]) || empty($data[$model->alias][$fieldName]['name'])) {
  756.                     if (!empty($data[$model->alias][$model->primaryKey]) || !$options['default']) {
  757.                         unset($data[$model->alias][$fieldName]);
  758.                     } else {
  759.                         $data[$model->alias][$fieldName] = $options['default'];
  760.                     }
  761.                 }
  762.  
  763.                 //if the record is already saved in the database, set the existing file to be removed after the save is sucessfull
  764.                 if (!empty($data[$model->alias][$model->primaryKey])) {
  765.                     $this->_setFileToRemove($model, $fieldName, $options['thumbnailDir']);
  766.                 }
  767.  
  768.                 // save in encrypted folder if specified
  769.                 if ($options['encryptedFolder']) {
  770.                     // setup UUID as a unique folder name
  771.                     $options['dir'] .= DS . String::uuid();
  772.                 }
  773.  
  774.                 // Fix the filename, removing bad characters and avoiding from overwriting existing ones
  775.                 if ($options['default'] == true) {
  776.                     $this->_includeDefaultReplacement($options['default']);
  777.                 }
  778.                 $this->_fixName($model, $fieldName);
  779.                 // Also save the original image as uploadName if that option is not empty
  780.                 if (isset($options['uploadName']) && !empty($options['uploadName'])) {
  781.                     $saveAs = $options['dir'] . DS . $data[$model->alias][$options['uploadName']].'.'.$ext;
  782.                 } else {
  783.                     $saveAs = $options['dir'] . DS . $data[$model->alias][$fieldName]['name'];
  784.                 }
  785.  
  786.                 // Attempt to move uploaded file
  787.                 $copyResults = $this->_copyFileFromTemp($data[$model->alias][$fieldName]['tmp_name'], $saveAs);
  788.                 if ($copyResults !== true) {
  789.                     $result = array('return' => false, 'reason' => 'validation', 'extra' => array('field' => $fieldName, 'error' => $copyResults));
  790.                     continue;
  791.                 }
  792.  
  793.                 // If the file is an image, try to make the thumbnails
  794.                 if ((count($options['thumbsizes']) > 0) && count($options['allowedExt']) > 0 && in_array($data[$model->alias][$fieldName]['type'], $this->_imageTypes)) {
  795.                     $this->_createThumbnails($model, $data, $fieldName, $saveAs, $ext, $options);
  796.                 }
  797.  
  798.                 // Update model data
  799.                 $data[$model->alias][$options['fields']['dir']] = $options['dir'];
  800.                 $data[$model->alias][$options['fields']['mimetype']] = $data[$model->alias][$fieldName]['type'];
  801.                 $data[$model->alias][$options['fields']['filesize']] = $data[$model->alias][$fieldName]['size'];
  802.                 if (isset($options['uploadName']) && !empty($options['uploadName'])) {
  803.                     $data[$model->alias][$fieldName] = $data[$model->alias][$options['uploadName']].'.'.$ext;
  804.                 } else {
  805.                     $data[$model->alias][$fieldName] = $data[$model->alias][$fieldName]['name'];
  806.                 }
  807.                 $result = array('return' => true, 'data' => $data);
  808.                 continue;
  809.             }
  810.         }
  811.         if (isset($result)) {
  812.             return $result;
  813.         } else {
  814.             return true;
  815.         }
  816.     }
  817.  
  818. /**
  819.  * Create all the thumbnails
  820.  *
  821.  * @param object $model Reference to model
  822.  * @param array $data 
  823.  * @param string $fieldName 
  824.  * @param string $ext 
  825.  * @param array $options 
  826.  * @return void 
  827.  * @access protected
  828.  */
  829.     function _createThumbnails(&$model, $data, $fieldName, $saveAs, $ext, $options) {
  830.         foreach ($options['thumbsizes'] as $key => $value) {
  831.             // Generate the name for the thumbnail
  832.             if (isset($options['uploadName']) && !empty($options['uploadName'])) {
  833.                 $thumbSaveAs = $this->_getThumbnailName($saveAs, $options['dir'], $options['thumbnailDir'], $key, $data[$model->alias][$options['uploadName']], $ext);
  834.             } else {
  835.                 $thumbSaveAs = $this->_getThumbnailName($saveAs, $options['dir'], $options['thumbnailDir'], $key, $data[$model->alias][$fieldName]['name']);
  836.             }
  837.             $params = array();
  838.             if (isset($value['width'])) {
  839.                 $params['thumbWidth'] = $value['width'];
  840.             }
  841.             if (isset($value['height'])) {
  842.                 $params['thumbHeight'] = $value['height'];
  843.             }
  844.             if (isset($value['maxDimension'])) {
  845.                 $params['maxDimension'] = $value['maxDimension'];
  846.             }
  847.             if (isset($value['thumbnailQuality'])) {
  848.                 $params['thumbnailQuality'] = $value['thumbnailQuality'];
  849.             }
  850.             if (isset($value['zoomCrop'])) {
  851.                 $params['zoomCrop'] = $value['zoomCrop'];
  852.             }
